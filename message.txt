    <script>
        // ------------- DATA MODEL --------------
        let tasks = [];
        let currentView = 'daily';
        let editingTaskId = null;

        // DOM elements
        const taskListEl = document.getElementById('taskList');
        const taskInput = document.getElementById('taskInput');
        const addBtn = document.getElementById('addBtn');
        const totalSpan = document.getElementById('totalCount');
        const completedSpan = document.getElementById('completedCount');
        const pendingSpan = document.getElementById('pendingCount');
        const currentDateDisplay = document.getElementById('currentDateDisplay');
        const currentTimeDisplay = document.getElementById('currentTimeDisplay');
        const darkModeToggle = document.getElementById('darkModeToggle');
        const scopeHint = document.getElementById('scopeHint');

        // Helper: format timestamp nicely
        function formatTimestamp(dateISO) {
            if (!dateISO) return 'Unknown date';
            const date = new Date(dateISO);
            if (isNaN(date.getTime())) return 'Invalid date';
            return date.toLocaleString('en-US', {
                year: 'numeric',
                month: 'short',
                day: 'numeric',
                hour: '2-digit',
                minute: '2-digit',
                second: '2-digit'
            });
        }

        // Load from localStorage
        function loadTasks() {
            const stored = localStorage.getItem('enhanced_tasks_v4_timestamp');
            if (stored) {
                tasks = JSON.parse(stored);
                // Ensure each task has createdAt if missing (legacy)
                tasks.forEach(task => {
                    if (!task.createdAt) {
                        task.createdAt = new Date().toISOString();
                    }
                });
            } else {
                // Demo tasks with realistic timestamps (some older, some recent)
                const now = new Date();
                const yesterday = new Date(now);
                yesterday.setDate(now.getDate() - 1);
                const twoDaysAgo = new Date(now);
                twoDaysAgo.setDate(now.getDate() - 2);
                
                tasks = [
                    { id: Date.now() + 1, text: "📝 Daily standup meeting", completed: false, scope: 'daily', createdAt: now.toISOString() },
                    { id: Date.now() + 2, text: "💼 Prepare weekly progress report", completed: false, scope: 'weekly', createdAt: yesterday.toISOString() },
                    { id: Date.now() + 3, text: "🎯 Achieve monthly sales target", completed: false, scope: 'monthly', createdAt: twoDaysAgo.toISOString() },
                    { id: Date.now() + 4, text: "✅ Review emails and prioritize", completed: true, scope: 'daily', createdAt: new Date(now.getTime() - 3 * 3600000).toISOString() }
                ];
            }
        }

        function saveTasks() {
            localStorage.setItem('enhanced_tasks_v4_timestamp', JSON.stringify(tasks));
        }

        function getTasksByCurrentView() {
            if (currentView === 'daily') return tasks.filter(task => task.scope === 'daily');
            else if (currentView === 'weekly') return tasks.filter(task => task.scope === 'weekly');
            else return tasks.filter(task => task.scope === 'monthly');
        }

        function updateStats() {
            const filtered = getTasksByCurrentView();
            const total = filtered.length;
            const completed = filtered.filter(t => t.completed).length;
            const pending = total - completed;
            totalSpan.innerText = total;
            completedSpan.innerText = completed;
            pendingSpan.innerText = pending;
        }

        // Add task with current timestamp
        function addTask() {
            const taskText = taskInput.value.trim();
            if (taskText === '') {
                alert('Please enter a task!');
                return;
            }
            const newTask = {
                id: Date.now(),
                text: taskText,
                completed: false,
                scope: currentView,
                createdAt: new Date().toISOString()   // 🕒 store exact date & time
            };
            tasks.push(newTask);
            saveTasks();
            taskInput.value = '';
            renderTasks();
        }

        function toggleTaskCompletion(id) {
            const task = tasks.find(t => t.id === id);
            if (task) {
                task.completed = !task.completed;
                saveTasks();
                renderTasks();
            }
        }

        function deleteTask(id) {
            if (confirm('Delete this task?')) {
                tasks = tasks.filter(t => t.id !== id);
                if (editingTaskId === id) editingTaskId = null;
                saveTasks();
                renderTasks();
            }
        }

        function startEdit(id) {
            editingTaskId = id;
            renderTasks();
        }

        function saveEdit(id, newText) {
            const trimmed = newText.trim();
            if (trimmed === '') {
                alert('Task text cannot be empty!');
                return;
            }
            const task = tasks.find(t => t.id === id);
            if (task) {
                task.text = trimmed;
                saveTasks();
            }
            editingTaskId = null;
            renderTasks();
        }

        function cancelEdit() {
            editingTaskId = null;
            renderTasks();
        }

        function escapeHtml(str) {
            if (!str) return '';
            return str.replace(/[&<>]/g, function(m) {
                if (m === '&') return '&amp;';
                if (m === '<') return '&lt;';
                if (m === '>') return '&gt;';
                return m;
            });
        }

        // Render tasks with timestamp display
        function renderTasks() {
            const filteredTasks = getTasksByCurrentView();
            const container = taskListEl;
            
            if (filteredTasks.length === 0) {
                let emptyMessage = '';
                if (currentView === 'daily') emptyMessage = '✨ No daily tasks. Add a task for today!';
                else if (currentView === 'weekly') emptyMessage = '📅 No weekly tasks. Plan your week!';
                else emptyMessage = '📆 No monthly tasks. Set your monthly goals!';
                container.innerHTML = `<div class="empty-state"><i class="fas fa-clipboard-list fa-3x mb-3"></i><p>${emptyMessage}</p></div>`;
                updateStats();
                return;
            }
            
            container.innerHTML = filteredTasks.map(task => {
                const isEditing = (editingTaskId === task.id);
                const statusLabel = task.completed ? 'Done' : 'Pending';
                const statusClass = task.completed ? 'status-done' : 'status-pending';
                const timeCreated = formatTimestamp(task.createdAt);
                
                if (isEditing) {
                    return `
                        <div class="task-item" data-id="${task.id}">
                            <div class="task-main-row">
                                <div class="task-content" style="flex:1;">
                                    <input type="text" id="edit-input-${task.id}" class="edit-input" value="${escapeHtml(task.text)}" autofocus>
                                </div>
                                <div class="task-actions">
                                    <button class="save-edit-btn" onclick="window.saveEdit(${task.id}, document.getElementById('edit-input-${task.id}').value)">
                                        <i class="fas fa-check me-1"></i>Save
                                    </button>
                                    <button class="cancel-edit-btn" onclick="window.cancelEdit()">
                                        <i class="fas fa-times me-1"></i>Cancel
                                    </button>
                                </div>
                            </div>
                            <div class="timestamp">
                                <i class="far fa-calendar-plus"></i> Created: ${escapeHtml(timeCreated)}
                            </div>
                        </div>
                    `;
                } else {
                    return `
                        <div class="task-item">
                            <div class="task-main-row">
                                <div class="task-content">
                                    <input type="checkbox" ${task.completed ? 'checked' : ''} onchange="window.toggleTaskCompletion(${task.id})">
                                    <div class="task-text-wrapper">
                                        <div class="task-text ${task.completed ? 'completed' : ''}" ondblclick="window.startEdit(${task.id})" title="Double-click to edit">
                                            ${escapeHtml(task.text)}
                                        </div>
                                        <div class="timestamp">
                                            <i class="far fa-calendar-plus"></i> Created: ${escapeHtml(timeCreated)}
                                        </div>
                                    </div>
                                    <span class="status-badge ${statusClass}">
                                        <i class="fas ${task.completed ? 'fa-check-circle' : 'fa-clock'} me-1"></i>${statusLabel}
                                    </span>
                                </div>
                                <div class="task-actions">
                                    <button class="edit-btn" onclick="window.startEdit(${task.id})" title="Edit task">
                                        <i class="fas fa-pen me-1"></i>Edit
                                    </button>
                                    <button class="delete-btn" onclick="window.deleteTask(${task.id})" title="Delete task">
                                        <i class="fas fa-trash me-1"></i>Del
                                    </button>
                                </div>
                            </div>
                        </div>
                    `;
                }
            }).join('');
            updateStats();
        }

        function setView(view) {
            currentView = view;
            editingTaskId = null;
            
            document.querySelectorAll('.view-btn').forEach(btn => {
                const btnView = btn.getAttribute('data-view');
                if (btnView === view) {
                    btn.classList.add('active');
                    btn.classList.remove('btn-outline-primary');
                    btn.classList.add('btn-primary');
                } else {
                    btn.classList.remove('active', 'btn-primary');
                    btn.classList.add('btn-outline-primary');
                }
            });
            
            if (view === 'daily') scopeHint.innerText = '🌞 Tasks for today (Daily) - timestamp shows when you added';
            else if (view === 'weekly') scopeHint.innerText = '📅 Tasks for this week (Weekly) - timestamp shows when you added';
            else scopeHint.innerText = '📆 Tasks for this month (Monthly) - timestamp shows when you added';
            
            renderTasks();
        }

        // Live Date & Time
        function updateDateTime() {
            const now = new Date();
            const optionsDate = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
            const formattedDate = now.toLocaleDateString('en-US', optionsDate);
            const formattedTime = now.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit', second: '2-digit' });
            currentDateDisplay.innerHTML = `<i class="far fa-calendar-alt me-1"></i> ${formattedDate}`;
            currentTimeDisplay.innerHTML = `<i class="far fa-clock me-1"></i> ${formattedTime}`;
        }

        // Dark Mode
        function initDarkMode() {
            const savedTheme = localStorage.getItem('todo_theme');
            if (savedTheme === 'dark') {
                document.body.classList.add('dark-mode');
                darkModeToggle.innerHTML = '<i class="fas fa-sun me-1"></i> Light Mode';
            } else {
                document.body.classList.remove('dark-mode');
                darkModeToggle.innerHTML = '<i class="fas fa-moon me-1"></i> Dark Mode';
            }
        }
        
        function toggleDarkMode() {
            if (document.body.classList.contains('dark-mode')) {
                document.body.classList.remove('dark-mode');
                localStorage.setItem('todo_theme', 'light');
                darkModeToggle.innerHTML = '<i class="fas fa-moon me-1"></i> Dark Mode';
            } else {
                document.body.classList.add('dark-mode');
                localStorage.setItem('todo_theme', 'dark');
                darkModeToggle.innerHTML = '<i class="fas fa-sun me-1"></i> Light Mode';
            }
        }

        // Legacy migration
        function migrateLegacyTasks() {
            let needsSave = false;
            tasks.forEach(task => {
                if (!task.scope) {
                    task.scope = 'daily';
                    needsSave = true;
                }
                if (task.completed === undefined) {
                    task.completed = false;
                    needsSave = true;
                }
                if (!task.createdAt) {
                    task.createdAt = new Date().toISOString();
                    needsSave = true;
                }
            });
            if (needsSave) saveTasks();
        }

        function attachEventListeners() {
            addBtn.onclick = () => addTask();
            taskInput.addEventListener('keypress', (e) => {
                if (e.key === 'Enter') addTask();
            });
            darkModeToggle.onclick = toggleDarkMode;
            
            document.querySelectorAll('.view-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const view = btn.getAttribute('data-view');
                    if (view === 'daily') setView('daily');
                    else if (view === 'weekly') setView('weekly');
                    else if (view === 'monthly') setView('monthly');
                });
            });
        }

        // Expose globals
        window.toggleTaskCompletion = toggleTaskCompletion;
        window.deleteTask = deleteTask;
        window.startEdit = startEdit;
        window.saveEdit = saveEdit;
        window.cancelEdit = cancelEdit;
        window.addTask = addTask;

        function init() {
            loadTasks();
            migrateLegacyTasks();
            initDarkMode();
            attachEventListeners();
            setView('daily');
            updateDateTime();
            setInterval(updateDateTime, 1000);
            renderTasks();
        }
        
        init();
    </script>
</body>
